# Development-only MP bounded payload PoC.
# Pure Stellaris script does not currently expose a verified is_host trigger, so
# this is a single-authority approximation: exactly one human country writes a
# tiny predefined payload into synchronized game state, then other human
# countries observe that state on their monthly pulse.
country_event = {
    id = strategic_nexus.1300
    title = strategic_nexus.1300.name
    desc = strategic_nexus.1300.desc
    picture = GFX_evt_space_station
    show_sound = event_default
    is_triggered_only = yes

    trigger = {
        is_multiplayer = yes
        is_ai = no
        is_country_type = default
        NOT = { has_global_flag = strategic_nexus_poc_mp_payload_generated }
    }

    immediate = {
        set_global_flag = strategic_nexus_poc_mp_payload_generated
        set_global_flag = strategic_nexus_bridge_available
        set_global_flag = strategic_nexus_doctrine_balance_against_hegemon
        set_global_flag = strategic_nexus_military_posture_defensive
        set_global_flag = strategic_nexus_research_bias_economy
        set_country_flag = strategic_nexus_poc_mp_payload_authority
        set_country_flag = strategic_nexus_bridge_available
        set_country_flag = strategic_nexus_doctrine_balance_against_hegemon
        set_country_flag = strategic_nexus_military_posture_defensive
        set_country_flag = strategic_nexus_research_bias_economy
        set_variable = {
            which = strategic_nexus_pressure
            value = 0.74
        }

        log = "Strategic Nexus: MP bounded payload generated once by script authority"
        country_event = { id = strategic_nexus.1002 }
        country_event = { id = strategic_nexus.1400 }
    }

    option = {
        name = strategic_nexus.1300.a
    }
}

# Non-Ironman v0.x PoC only.
# This event verifies whether bounded flags/variables can be detected by ordinary Stellaris mod script.
country_event = {
    id = strategic_nexus.1002
    title = strategic_nexus.1002.name
    desc = strategic_nexus.1002.desc
    picture = GFX_evt_space_station
    show_sound = event_default
    is_triggered_only = yes

    immediate = {
        remove_country_flag = strategic_nexus_poc_event_seen
        remove_country_flag = strategic_nexus_poc_fallback_seen
        remove_country_flag = strategic_nexus_poc_global_bridge_available_seen
        remove_country_flag = strategic_nexus_poc_country_bridge_available_seen
        remove_country_flag = strategic_nexus_poc_balance_doctrine_seen
        remove_country_flag = strategic_nexus_poc_pressure_variable_seen
        remove_modifier = strategic_nexus_poc_bridge_seen

        set_country_flag = strategic_nexus_poc_event_seen

        if = {
            limit = {
                NOR = {
                    has_global_flag = strategic_nexus_bridge_available
                    has_country_flag = strategic_nexus_bridge_available
                }
            }
            log = "Strategic Nexus: fallback path selected, no bridge availability flag detected"
            set_country_flag = strategic_nexus_poc_fallback_seen
            country_event = { id = strategic_nexus.1104 }
        }

        if = {
            limit = {
                has_global_flag = strategic_nexus_bridge_available
            }
            log = "Strategic Nexus: global bridge flag detected"
            set_country_flag = strategic_nexus_poc_global_bridge_available_seen
            country_event = { id = strategic_nexus.1100 }
        }

        if = {
            limit = {
                has_country_flag = strategic_nexus_bridge_available
            }
            log = "Strategic Nexus: country bridge flag detected"
            set_country_flag = strategic_nexus_poc_country_bridge_available_seen
            country_event = { id = strategic_nexus.1101 }
        }

        if = {
            limit = {
                OR = {
                    has_global_flag = strategic_nexus_doctrine_balance_against_hegemon
                    has_country_flag = strategic_nexus_doctrine_balance_against_hegemon
                }
            }
            log = "Strategic Nexus: doctrine balance against hegemon flag detected"
            set_country_flag = strategic_nexus_poc_balance_doctrine_seen
            country_event = { id = strategic_nexus.1102 }
        }

        if = {
            limit = {
                check_variable = {
                    which = strategic_nexus_pressure
                    value > 0.5
                }
            }
            log = "Strategic Nexus: strategic pressure variable detected above threshold"
            set_country_flag = strategic_nexus_poc_pressure_variable_seen
            country_event = { id = strategic_nexus.1103 }
        }

        # Harmless diagnostic marker. This is intentionally tiny and temporary.
        # It exists only to make the PoC visible in non-Ironman v0.x testing.
        if = {
            limit = {
                has_country_flag = strategic_nexus_poc_balance_doctrine_seen
            }
            add_modifier = {
                modifier = strategic_nexus_poc_bridge_seen
                days = 360
            }
            log = "Strategic Nexus: diagnostic bridge modifier applied"
            country_event = { id = strategic_nexus.1105 }
        }
    }

    option = {
        name = strategic_nexus.1002.a
    }
}

# Development-only generated overlay projection.
# This consumes generated v0 preference flags and makes the selected branch
# visible as a harmless country modifier after start/load without adding hidden
# gameplay buffs.
country_event = {
    id = strategic_nexus.1450
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        is_ai = no
        is_country_type = default
    }

    immediate = {
        # Stable handwritten monthly kernel touchpoint for allowlisted generated
        # reactive policy-pack branches. The generated overlay owns the actual
        # per-rule dispatch behind this effect name.
        strategic_nexus_generated_monthly_strategy_tick_dispatch = yes

        remove_modifier = strategic_nexus_poc_generated_military_posture_defensive
        remove_modifier = strategic_nexus_poc_generated_military_posture_aggressive
        remove_modifier = strategic_nexus_poc_generated_research_bias_economy
        remove_modifier = strategic_nexus_poc_generated_research_bias_military_industry

        if = {
            limit = {
                has_country_flag = strategic_nexus_pref_military_posture_defensive
                NOT = { has_country_flag = strategic_nexus_pref_military_posture_aggressive }
            }
            add_modifier = {
                modifier = strategic_nexus_poc_generated_military_posture_defensive
                days = 40
            }
            log = "Strategic Nexus: generated overlay projected military_posture defensive marker"
        }

        if = {
            limit = {
                has_country_flag = strategic_nexus_pref_military_posture_aggressive
                NOT = { has_country_flag = strategic_nexus_pref_military_posture_defensive }
            }
            add_modifier = {
                modifier = strategic_nexus_poc_generated_military_posture_aggressive
                days = 40
            }
            log = "Strategic Nexus: generated overlay projected military_posture aggressive marker"
        }

        if = {
            limit = {
                has_country_flag = strategic_nexus_pref_research_bias_economy
                NOT = { has_country_flag = strategic_nexus_pref_research_bias_military_industry }
            }
            add_modifier = {
                modifier = strategic_nexus_poc_generated_research_bias_economy
                days = 40
            }
            log = "Strategic Nexus: generated overlay projected research_bias economy marker"
        }

        if = {
            limit = {
                has_country_flag = strategic_nexus_pref_research_bias_military_industry
                NOT = { has_country_flag = strategic_nexus_pref_research_bias_economy }
            }
            add_modifier = {
                modifier = strategic_nexus_poc_generated_research_bias_military_industry
                days = 40
            }
            log = "Strategic Nexus: generated overlay projected research_bias military_industry marker"
        }
    }
}
